045/**
046 * Test that the SecurityService works properly by comparing behaviour of Turbine and Fulcrum security services using memory user manager.
047 *
048 * Code adapted from SecurityServiceAdapter in Fulcrum Security Adapter
049 *
050 * @author gkallidis
051 * @version $Id$
052 */
053
054public class SecurityServiceTest extends BaseUnit5Test
055{
056
057    SecurityService fulcrumSecurityService;
058    org.apache.turbine.services.security.SecurityService securityService;
059    static TurbineConfig tc;
060
061
062    @BeforeAll
063    public static void init() throws Exception
064    {
065        tc = new TurbineConfig(".", "/conf/test/CompleteTurbineResources.properties");
066        tc.initialize();
067
068    }
069
070    @BeforeEach
071    public void setUpBefore() throws Exception
072    {
073
074        ServiceManager serviceManager = TurbineServices.getInstance();
075        //
076        fulcrumSecurityService = (SecurityService) serviceManager.getService(SecurityService.ROLE);
077
078        securityService = (org.apache.turbine.services.security.SecurityService)
079                        TurbineServices.getInstance().getService(org.apache.turbine.services.security.SecurityService.SERVICE_NAME);
080    }
081
082    @Test
083    public void testAccountExists() throws Exception
084    {
085
086        User user = new org.apache.turbine.om.security.DefaultUserImpl(new TurbineUserImpl());
087        user.setAccessCounter(5);
088
089        assertFalse(securityService.accountExists(user));
090        assertFalse(fulcrumSecurityService.getUserManager().checkExists(user));
091
092    }
093    @Test
094        public void testCreateUser() throws Exception
095        {
096
097                User user = new org.apache.turbine.om.security.DefaultUserImpl(new TurbineUserImpl());
098                user.setAccessCounter(5);
099                user.setName("ringo");
100                securityService.addUser(user,"fakepasswrod");
101                assertTrue(securityService.accountExists(user));
102                assertTrue(fulcrumSecurityService.getUserManager().checkExists(user));
103
104        }
105
106    /**
107     * Tests Turbine and Fulcrum.
108     *
109     * Fulcrum part is similar/duplicated from {@link AbstractTurbineModelManagerTest#testGrantUserGroupRole()}
110     *
111     *
112     * @throws Exception
113     */
114    @Test
115    public void testGrantUserGroupRole() throws Exception
116    {
117        Group group = fulcrumSecurityService.getGroupManager().getGroupInstance();
118        group.setName("TEST_GROUP");
119        fulcrumSecurityService.getGroupManager().addGroup(group);
120        Role role = fulcrumSecurityService.getRoleManager().getRoleInstance();
121        role.setName("TEST_Role");
122        fulcrumSecurityService.getRoleManager().addRole(role);
123
124        //  Turbine security service returns a wrapped instance: org.apache.turbine.om.security.DefaultUserImpl
125        // which implements org.apache.turbine.om.security.User and contains
126        User user = securityService.getUserInstance("Clint");
127                // memory
128        securityService.addUser(user, "clint");
129        securityService.grant(user, group, role);
130
131                addUserAndCheck(group, role, user.getUserDelegate());
132
133        // Fulcrum security service returns a raw org.apache.fulcrum.security.model.turbine.entity.impl.TurbineUserImpl,
134                org.apache.fulcrum.security.UserManager  userManager = fulcrumSecurityService.getUserManager();
135                TurbineUser fulcrumUser = userManager.getUserInstance("Clint2");
136        userManager.addUser(fulcrumUser, "clint2");         // memory
137        ((TurbineModelManager)fulcrumSecurityService.getModelManager()).grant(fulcrumUser, group, role);
138
139        addUserAndCheck(group, role, fulcrumUser);
140
141    }
142
143    /**
144     * Fulcrum contract check
145     *
146     * @param group Fulcrum interface
147     * @param role Fulcrum interface
148     * @param user Fulcrum interface
149     * @throws EntityExistsException
150     * @throws DataBackendException
151     * @throws UnknownEntityException
152     */
153        private void addUserAndCheck(Group group, Role role, TurbineUser user)
154                        throws EntityExistsException, DataBackendException,
155                        UnknownEntityException
156        {
157
158        boolean ugrFound = false;
159        TurbineUserGroupRole ugrTest = null;
160        for (TurbineUserGroupRole ugr : user.getUserGroupRoleSet())
161        {
162            if (ugr.getUser().equals(user) && ugr.getGroup().equals(group) && ugr.getRole().equals(role))
163            {
164                ugrFound = true;
165                ugrTest = ugr;
166                break;
167            }
168        }
169        assertTrue(ugrFound);
170        assertTrue(ugrTest.getGroup().equals(group));
171        assertTrue(ugrTest.getUser().equals(user));
172        }
173
174
175        @AfterAll
176        public static void setupAfter()
177    {
178        ServiceManager serviceManager = TurbineServices.getInstance();
179        serviceManager.shutdownService(org.apache.turbine.services.security.SecurityService.SERVICE_NAME);
180        serviceManager.shutdownServices();
181    }
