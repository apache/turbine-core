024import org.apache.avalon.framework.logger.AbstractLogEnabled;
025import org.apache.avalon.framework.activity.Initializable;
026import org.apache.avalon.framework.activity.Disposable;
027import org.apache.avalon.framework.context.Contextualizable;
028import org.apache.avalon.framework.context.Context;
029import org.apache.avalon.framework.context.ContextException;
030
031/**
032 * Implementation of the test component.
033 *
034 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
035 * @version $Id$
036 */
037public class TestComponentImpl
038        extends AbstractLogEnabled
039        implements Initializable, Disposable, TestComponent, Contextualizable
040{
041    private String appRoot;
042
043    @Override
044    public void initialize() throws Exception
045    {
046        // empty
047    }
048
049    @Override
050    public void dispose()
051    {
052        // empty
053    }
054
055    @Override
056    public void test()
057    {
058        setupLogger(this, "TestComponent");
059        getLogger().debug("test");
060        getLogger().debug("componentAppRoot = "+appRoot);
061    }
062
063    @Override
064    public void contextualize(Context context) throws ContextException
065    {
066        appRoot = (String) context.get("componentAppRoot");
067    }
068}
