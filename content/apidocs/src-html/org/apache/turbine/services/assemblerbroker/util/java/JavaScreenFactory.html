024import org.apache.turbine.modules.Loader;
025import org.apache.turbine.modules.Screen;
026import org.apache.turbine.modules.ScreenLoader;
027
028/**
029 * A screen factory that attempts to load a java class from
030 * the module packages defined in the TurbineResource.properties.
031 *
032 * @author <a href="mailto:leon@opticode.co.za">Leon Messerschmidt</a>
033 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
034 * @version $Id$
035 */
036public class JavaScreenFactory
037        extends JavaBaseFactory<Screen>
038{
039    /**
040     * Get an Assembler.
041     *
042     * @param name name of the requested Assembler
043     * @return an Assembler
044     */
045    @Override
046    public Screen getAssembler(String name)
047    {
048        return getAssembler(Screen.PREFIX, name);
049    }
050
051    /**
052     * Get the loader for this type of assembler
053     *
054     * @return a Loader
055     */
056    @Override
057    public Loader<Screen> getLoader()
058    {
059        return ScreenLoader.getInstance();
060    }
061
062    /**
063     * Get the class of this assembler
064     *
065     * @return a class
066     */
067    @Override
068    public Class<Screen> getManagedClass()
069    {
070        return Screen.class;
071    }
072}
