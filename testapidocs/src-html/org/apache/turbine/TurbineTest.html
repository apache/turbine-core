022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertNotNull;
024
025import javax.servlet.http.HttpServletRequest;
026import javax.servlet.http.HttpServletResponse;
027
028import org.apache.turbine.test.BaseTestCase;
029import org.apache.turbine.util.TurbineConfig;
030import org.junit.After;
031import org.junit.Before;
032import org.junit.Test;
033import org.mockito.Mockito;
034
035/**
036 * This testcase verifies that TurbineConfig can be used to startup Turbine in a
037 * non servlet environment properly.
038 *
039 * @author <a href="mailto:epugh@opensourceconnections.com">Eric Pugh </a>
040 * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux </a>
041 * @version $Id$
042 */
043public class TurbineTest extends BaseTestCase
044{
045    private TurbineConfig tc = null;
046
047    @Before
048    public void setUp() throws Exception
049    {
050        tc = new TurbineConfig(".",
051                "/conf/test/CompleteTurbineResources.properties");
052        tc.initialize();
053    }
054
055    @After
056    public void tearDown() throws Exception
057    {
058        if (tc != null)
059        {
060            tc.dispose();
061        }
062    }
063
064    @Test
065    public void testTurbineAndFirstGet() throws Exception
066    {
067        assertNotNull(Turbine.getDefaultServerData());
068        assertEquals("", Turbine.getServerName());
069        assertEquals("80", Turbine.getServerPort());
070        assertEquals("", Turbine.getScriptName());
071        Turbine t = tc.getTurbine();
072
073        HttpServletRequest request = getMockRequest();
074        HttpServletResponse resp = Mockito.mock(HttpServletResponse.class);
075
076        t.doGet(request, resp);
077
078        assertEquals("8080", Turbine.getServerPort());
079        t.destroy();
080    }
081
082    @Test
083    @Deprecated
084    public void testDefaultInputEncoding() throws Exception
085    {
086        Turbine t = tc.getTurbine();
087        assertNotNull(Turbine.getDefaultInputEncoding());
088        assertEquals(TurbineConstants.PARAMETER_ENCODING_DEFAULT, Turbine.getDefaultInputEncoding());
089        t.destroy();
090    }
091
092    @Test
093    @Deprecated
094    public void testNonDefaultEncoding()
095    {
096        Turbine t = tc.getTurbine();
097        Turbine.getConfiguration().setProperty(TurbineConstants.PARAMETER_ENCODING_KEY, "UTF-8");
098        assertNotNull(Turbine.getDefaultInputEncoding());
099        assertEquals("UTF-8", Turbine.getDefaultInputEncoding());
100        t.destroy();
101    }
102}
