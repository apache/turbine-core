022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertFalse;
024import static org.junit.Assert.assertTrue;
025import static org.junit.Assert.fail;
026import static org.junit.jupiter.api.Assertions.assertInstanceOf;
027
028import org.apache.turbine.modules.scheduledjobs.SimpleJob;
029import org.apache.turbine.services.TurbineServices;
030import org.apache.turbine.test.BaseTestCase;
031import org.apache.turbine.util.TurbineConfig;
032import org.apache.turbine.util.TurbineException;
033import org.junit.After;
034import org.junit.Before;
035import org.junit.Test;
036import org.quartz.JobKey;
037
038/**
039 * Unit testing for the quartz implementation of the scheduler service.
040 *
041 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
042 */
043public class QuartzSchedulerServiceTest extends BaseTestCase
044{
045    private TurbineConfig tc = null;
046
047    private ScheduleService scheduler = null;
048
049    @Before
050    public void setUp() throws Exception
051    {
052        tc =
053            new TurbineConfig(
054                ".",
055                "/conf/test/TestFulcrumComponents.properties");
056        tc.initialize();
057
058        scheduler = (ScheduleService)TurbineServices.getInstance().getService(ScheduleService.SERVICE_NAME);
059    }
060
061    @After
062    public void tearDown() throws Exception
063    {
064        if (tc != null)
065        {
066            tc.dispose();
067        }
068    }
069
070    /**
071     * Tests the ability to enable and disable the service.
072     */
073    @Test public void testEnableDisable()
074    {
075        try
076        {
077            scheduler.startScheduler();
078            assertTrue(scheduler.isEnabled());
079
080            scheduler.stopScheduler();
081            assertFalse(scheduler.isEnabled());
082        }
083        catch (Exception e)
084        {
085            e.printStackTrace();
086            fail();
087        }
088    }
089
090    /**
091     * Tests the ability to add and remove a job.  A list of jobs will be obtained from
092     * the service to determine if the operation were successful.
093     */
094    @Test public void testAddRemoveJob()
095    {
096        try
097        {
098            // get the current job count for later comparison
099            int jobCount = scheduler.listJobs().size();
100
101            // Add a new job entry
102                        JobEntry je = scheduler.newJob(10, -1, -1, -1, -1, "SimpleJob1");
103            je.setJobId(jobCount + 1);
104
105            scheduler.addJob(je);
106            assertEquals(jobCount + 1, scheduler.listJobs().size());
107
108            scheduler.removeJob(je);
109            assertEquals(jobCount, scheduler.listJobs().size());
110
111        }
112        catch (Exception e)
113        {
114            e.printStackTrace();
115            fail();
116        }
117    }
118
119    /**
120     * Tests the ability to retrieve the job added during initialization.
121     */
122    @Test public void testGetJob()
123    {
124        try
125        {
126            JobKey jk = new JobKey("SimpleJob", JobEntryQuartz.DEFAULT_JOB_GROUP_NAME);
127                        JobEntry je = scheduler.getJob(jk.hashCode());
128                        assertInstanceOf(JobEntryQuartz.class, je);
129                        JobEntryQuartz jeq = (JobEntryQuartz)je;
130            assertEquals(jeq.getJobTrigger().getJobKey(), jk);
131            assertEquals(jeq.getTask(), "SimpleJob");
132        }
133        catch (TurbineException e)
134        {
135            e.printStackTrace();
136            fail();
137        }
138    }
139
140    /**
141     * Test to make sure a job actually runs.
142     */
143    @Test public void testRunningJob()
144    {
145        try
146        {
147           int beforeCount = SimpleJob.getCounter();
148           Thread.sleep(1200);
149           int afterCount = SimpleJob.getCounter();
150           assertTrue(beforeCount < afterCount);
151
152        }
153        catch (Exception e)
154        {
155            e.printStackTrace();
156            fail();
157        }
158    }
159
160}
