022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertNotEquals;
024import static org.junit.Assert.assertNotNull;
025import static org.junit.Assert.assertTrue;
026
027import java.io.FileInputStream;
028import java.io.InputStream;
029import java.util.List;
030import java.util.Map;
031import java.util.regex.Pattern;
032
033import javax.xml.bind.JAXBContext;
034import javax.xml.bind.Unmarshaller;
035
036import org.junit.Before;
037import org.junit.Test;
038
039public class URLMappingContainerTest
040{
041    URLMappingContainer container;
042
043    @Before
044    public void setUp() throws Exception
045    {
046        try (InputStream reader = new FileInputStream("conf/turbine-url-mapping.xml"))
047        {
048            JAXBContext jaxb = JAXBContext.newInstance(URLMappingContainer.class);
049            Unmarshaller unmarshaller = jaxb.createUnmarshaller();
050            container = (URLMappingContainer) unmarshaller.unmarshal(reader);
051        }
052    }
053
054    @Test
055    public void testGetName()
056    {
057        assertNotNull(container);
058        assertEquals("default", container.getName());
059    }
060
061    @Test
062    public void testGetMapEntries()
063    {
064        assertNotNull(container);
065
066        List<URLMapEntry> mapEntries = container.getMapEntries();
067        assertNotNull(mapEntries);
068        assertNotEquals(0, mapEntries.size());
069
070        URLMapEntry entry = mapEntries.get(0);
071        assertNotNull(entry);
072
073        Pattern pattern = entry.getUrlPattern();
074        assertNotNull(pattern);
075        assertTrue(pattern.matcher("/app/book/123").matches());
076
077        Map<String, String> implicit = entry.getImplicitParameters();
078        assertNotNull(implicit);
079        assertEquals(2, implicit.size());
080        assertEquals("Book.vm", implicit.get("template"));
081        assertEquals("0", implicit.get("detail"));
082    }
083
084}
