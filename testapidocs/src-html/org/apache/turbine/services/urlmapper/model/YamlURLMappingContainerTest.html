003import static org.junit.jupiter.api.Assertions.assertEquals;
004import static org.junit.jupiter.api.Assertions.assertNotEquals;
005import static org.junit.jupiter.api.Assertions.assertNotNull;
006import static org.junit.jupiter.api.Assertions.assertTrue;
007
008import java.io.FileInputStream;
009import java.io.InputStream;
010import java.util.List;
011import java.util.Map;
012import java.util.regex.Pattern;
013
014import org.junit.jupiter.api.BeforeAll;
015import org.junit.jupiter.api.Tag;
016import org.junit.jupiter.api.Test;
017
018import com.fasterxml.jackson.databind.ObjectMapper;
019import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
020
021@Tag("yaml")
022public class YamlURLMappingContainerTest
023{
024    private static URLMappingContainer container;
025
026    @BeforeAll
027    public static void setUp() throws Exception
028    {
029        try (InputStream reader = new FileInputStream( "conf/turbine-url-mapping.yml" ))
030        {
031            ObjectMapper mapper = new ObjectMapper( new YAMLFactory() );
032            // List<URLMapEntry> urlList =
033            // mapper.readValue(reader, mapper.getTypeFactory().constructCollectionType(List.class,
034            // URLMapEntry.class));//
035            container = mapper.readValue( reader, URLMappingContainer.class );
036        }
037    }
038
039    @Test
040    public void testGetName()
041    {
042        assertNotNull( container );
043        assertEquals( "default", container.getName() );
044    }
045
046    @Test
047    public void testGetMapEntries()
048    {
049        assertNotNull( container );
050
051        List<URLMapEntry> mapEntries = container.getMapEntries();
052        assertNotNull( mapEntries );
053        assertNotEquals( 0, mapEntries.size() );
054
055        URLMapEntry entry = mapEntries.get( 0 );
056        assertNotNull( entry );
057
058        Pattern pattern = entry.getUrlPattern();
059        assertNotNull( pattern );
060        assertTrue( pattern.matcher( "/app/book/123" ).matches() );
061
062        Map<String, String> implicit = entry.getImplicitParameters();
063        assertNotNull( implicit );
064        assertEquals( 2, implicit.size() );
065        assertEquals( "Book.vm", implicit.get( "template" ) );
066        assertEquals( "0", implicit.get( "detail" ) );
067    }
068
069}
