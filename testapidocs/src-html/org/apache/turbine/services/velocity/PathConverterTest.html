024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertNotNull;
026
027import java.io.File;
028
029import org.apache.commons.configuration2.Configuration;
030import org.apache.turbine.Turbine;
031import org.apache.turbine.services.TurbineServices;
032import org.apache.turbine.test.BaseTestCase;
033import org.apache.turbine.util.TurbineConfig;
034import org.apache.velocity.app.VelocityEngine;

036import org.junit.BeforeClass;

039/**
040 * Tests startup of the Velocity Service and translation of various
041 * path patterns.
042 *
043 * @author <a href="hps@intermeta.de">Henning P. Schmiedehausen</a>
044 * @version $Id: PathConverterTest.java 1837261 2018-08-01 20:30:15Z tv $
045 */
046
047public class PathConverterTest
048    extends BaseTestCase
049{
050    private static TurbineConfig tc = null;
051    private static VelocityService vs = null;
052
053    @BeforeClass
054    public static void setUp() throws Exception {
055        tc = new TurbineConfig(".", "/conf/test/TemplateService.properties");
056        tc.initialize();
057
058        vs = (VelocityService) TurbineServices.getInstance().getService(VelocityService.SERVICE_NAME);
059    }
060
061    @AfterClass
062    public static void destroy() throws Exception {
063        vs.shutdown();
064        tc.dispose();
065    }
066
067    @Test public void testService()
068        throws Exception
069    {
070
071        // Can we start the service?
072        assertNotNull("Could not load Service!", vs);
073    }
074
075    @Test
076    public void testPathTranslation()
077        throws Exception
078    {
079        Configuration conf = vs.getConfiguration();
080        VelocityEngine ve = new VelocityEngine();
081        ((TurbineVelocityService) vs).setVelocityProperties(ve, conf);
082
083        String rootPath = Turbine.getRealPath("");
084
085        String test1 = (String) ve.getProperty("test1.resource.loader.path");
086        assertNotNull("No Test1 Property found", test1);
087        assertEquals("Test1 Path translation failed", rootPath
088                +File.separator+"relative"+File.separator+"path" , test1);
089
090        String test2 = (String) ve.getProperty("test2.resource.loader.path");
091        assertNotNull("No Test2 Property found", test2);
092        assertEquals("Test2 Path translation failed", rootPath
093                +File.separator+"absolute"+File.separator+"path" , test2);
094
095        String test3 = (String) ve.getProperty("test3.resource.loader.path");
096        assertNotNull("No Test3 Property found", test3);
097        assertEquals("Test3 Path translation failed", rootPath
098                +File.separator+"jar-file.jar!/", test3);
099
100        String test4 = (String) ve.getProperty("test4.resource.loader.path");
101        assertNotNull("No Test4 Property found", test4);
102        assertEquals("Test4 Path translation failed", rootPath
103                +File.separator+"jar-file.jar!/with/some/extensions" , test4);
104
105        String test5 = (String) ve.getProperty("test5.resource.loader.path");
106        assertNotNull("No Test5 Property found", test5);
107        assertEquals("Test5 Path translation failed", rootPath
108                +File.separator+"jar-file.jar" , test5);
109
110        String test6 = (String) ve.getProperty("test6.resource.loader.path");
111        assertNotNull("No Test6 Property found", test6);
112        assertEquals("Test6 Path translation failed", "jar:http://jar.on.website/" , test6);
113
114        String test7 = (String) ve.getProperty("test7.resource.loader.path");
115        assertNotNull("No Test7 Property found", test7);
116        assertEquals("Test7 Path translation failed", rootPath
117                +File.separator+"file"+File.separator
118                +"system"+File.separator+"reference" , test7);
119
120        String test8 = (String) ve.getProperty("test8.resource.loader.path");
121        assertNotNull("No Test8 Property found", test8);
122        assertEquals("Test8 Path translation failed", "http://reference.on.website/" , test8);
123    }
124}
