135    @TurbineRequiredRole({"user","admin"})
136    public void guardedMethoded() {
137        // do nothing
138    }
139
140    @Test
141    public void testRequiredRoleMethodProcess() throws Exception
142    {
143        RunData data = mock(RunData.class);
144        TurbineAccessControlList<?> acl = mock(TurbineAccessControlList.class);
145        Role role = mock(Role.class);
146        when(role.getName()).thenReturn( "user" );
147        // Group group = mock(Group.class);
148        when(acl.hasRole( role.getName() )).thenReturn( Boolean.TRUE );
149        when(data.getACL()).thenReturn(acl );
150
151        Method[] methods = getClass().getMethods();
152        for (Method m : methods) {
153            if (m.getName().equals( "guardedMethoded" )) {
154                assertTrue( AnnotationProcessor.isAuthorized( m, (TurbineAccessControlList)data.getACL(), ConditionType.ANY ));
155                assertFalse( AnnotationProcessor.isAuthorized( m, (TurbineAccessControlList)data.getACL(), ConditionType.COMPOUND ));
156            }
157        }
158    }

193        Role role = mock(Role.class);
194        when(role.getName()).thenReturn( "user" );
195        // Group group = mock(Group.class);
196        when(acl.hasRole( role.getName() )).thenReturn( Boolean.FALSE );
197        when(data.getACL()).thenReturn(acl );
198
199        Method[] methods = getClass().getMethods();
200        for (Method m : methods) {
201            if (m.getName().equals( "unguardedMethoded" )) {
202                // default is true, if not annotated
203                assertTrue( AnnotationProcessor.isAuthorized( m, (TurbineAccessControlList)data.getACL(), ConditionType.ANY ));
204            }
205        }
