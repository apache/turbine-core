150    @TurbineRequiredRole({"user","admin"})
151    public void guardedMethoded() {
152        // do nothing
153    }
154
155    @Test
156    public void testRequiredRoleMethodProcess() throws Exception
157    {
158        RunData data = mock(RunData.class);
159        TurbineAccessControlList<?> acl = mock(TurbineAccessControlList.class);
160        Role role = mock(Role.class);
161        when(role.getName()).thenReturn( "user" );
162        // Group group = mock(Group.class);
163        when(acl.hasRole( role.getName() )).thenReturn( Boolean.TRUE );
164        when(data.getACL()).thenReturn(acl );
165
166        Method[] methods = getClass().getMethods();
167        for (Method m : methods) {
168            if (m.getName().equals( "guardedMethoded" )) {
169                assertTrue( AnnotationProcessor.isAuthorized( m, (TurbineAccessControlList<?>)data.getACL(), ConditionType.ANY ));
170                assertFalse( AnnotationProcessor.isAuthorized( m, (TurbineAccessControlList<?>)data.getACL(), ConditionType.COMPOUND ));
171            }
172        }
173    }

215        for (Method m : methods) {

217                // default is true, if not annotated
