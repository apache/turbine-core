066    @Test
067    public void testCreateTurbineWithConfigurationXML() throws Exception
068    {
069        txc = new TurbineXmlConfig(".", "conf/test/TurbineConfiguration.xml");
070
071        try
072        {
073            txc.initialize();
074
075            Configuration configuration = Turbine.getConfiguration();
076            assertNotNull("No Configuration Object found!", configuration);
077            assertFalse("Make sure we have values", configuration.isEmpty());
078
079            // overridden value
080            String key = "module.cache";
081            assertEquals("Read a config value " + key + ", received:" + configuration.getString(key), "true", configuration.getString(key));
082
083            // non overridden value
084            key = "scheduledjob.cache.size";
085            assertEquals("Read a config value " + key + ", received:" + configuration.getString(key), "10", configuration.getString(key));
086        }
087        catch (Exception e)
088        {
089            throw e;
090        }
091        finally
092        {
093            txc.dispose();
094        }
095    }

170    @Test
171    public void testCreateTurbineWithXMLBuilderConfiguration() throws Exception
172    {
173        String configurationRessourcePath ="conf/test/ConfigurationBuilder.xml";
174        tc = new TurbineXmlConfig(".",configurationRessourcePath );
175
176        try
177        {
178            tc.initialize();
179
180            Configuration configuration = Turbine.getConfiguration();
181            assertNotNull("No Configuration Object found!", configuration);
182            assertFalse("Make sure we have values", configuration.isEmpty());
183
184            //assertTrue("Test  combined configuration is"+ configuration, configuration instanceof CombinedConfiguration);
185
186            // overridden value
187            String key = "scheduledjob.cache.size";
188            assertEquals("Read a config value " + key + ", received:" + configuration.getInt(key), 100, configuration.getInt(key));
189
190            // double overridden value
191            key = "module.cache";
192            assertEquals("Read a config value " + key + ", received:" + configuration.getBoolean(key), false, configuration.getBoolean(key));
193            // new property
194            key = "tests.test";
195            configuration.addProperty( key, 123 );
196            assertEquals("Read a config value " + key + ", received:" + configuration.getInt(key), 123, configuration.getInt(key));
197            // not set
198            key="test.nulltest3";
199            assertEquals("Read a included config value " + key + ", received:" + configuration.getString(key), null, configuration.getString(key));
200            // overridden value
201            key="services.PullService.earlyInit";
202            assertEquals("Read a config value " + key + ", received:" + configuration.getBoolean(key), true, configuration.getBoolean(key));
203            configuration.setProperty( key, false );
204            assertEquals("Read a config value " + key + ", received:" + configuration.getBoolean(key), false, configuration.getBoolean(key));
205
206            // converts to URL, cft. RFC2396
207            URL testURL = FileLocatorUtils.DEFAULT_FILE_SYSTEM.locateFromURL(new File( Turbine.getApplicationRoot()).toURI().toString() , configurationRessourcePath);
208            assertNotNull( "Should be a valid URL",testURL);
209        }
210        finally
211        {
212            tc.dispose();
213        }
214    }
