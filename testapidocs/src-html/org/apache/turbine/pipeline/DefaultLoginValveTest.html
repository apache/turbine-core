045import org.apache.turbine.util.TurbineConfig;

051/**
052 * Tests TurbinePipeline.
053 *
054 * @author <a href="mailto:epugh@opensourceConnections.com">Eric Pugh</a>
055 * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux</a>
056 * @version $Id$
057 */
058public class DefaultLoginValveTest extends BaseTestCase
059{
060    private static TurbineConfig tc = null;
061    private ServletConfig config = null;
062    private HttpServletRequest request = null;
063    private HttpServletResponse response = null;
064    private HttpSession session = null;
065    private SecurityService securityService = null;
066
067    @BeforeAll
068    public static void init()
069    {
070        tc = new TurbineConfig(
071                            ".",
072                            "/conf/test/CompleteTurbineResources.properties");
073        tc.initialize();
074    }
075
076    @BeforeEach
077    public void setUpBefore() throws Exception
078    {
079        config = mock(ServletConfig.class);
080        request = getMockRequest();
081        response = mock(HttpServletResponse.class);
082        session = request.getSession();
083
084        // User must exist
085        securityService = (SecurityService)TurbineServices.getInstance().getService(SecurityService.SERVICE_NAME);
086        if (!securityService.accountExists("username"))
087        {
088            User user = securityService.getUserInstance();
089            user.setName("username");
090            securityService.addUser(user, "password");
091        }
092    }
093
094    /**
095     * Tests the Valve.
096     */
097    @Test
098    public void testDefaults() throws Exception
099    {
100        Vector<String> v = new Vector<>();
101        v.add(LoginUser.CGI_USERNAME);
102        v.add(LoginUser.CGI_PASSWORD);
103        when(request.getParameterNames()).thenReturn(v.elements());
104
105        when(request.getParameterValues(LoginUser.CGI_USERNAME)).thenReturn(new String[] { "username" });
106        when(request.getParameterValues(LoginUser.CGI_PASSWORD)).thenReturn(new String[] { "password" });
107
108        RunData runData = getRunData(request,response,config);
109        runData.setAction(TurbineConstants.ACTION_LOGIN_DEFAULT);
110
111        Pipeline pipeline = new TurbinePipeline();
112        PipelineData pipelineData = runData;
113
114        DefaultLoginValve valve = new DefaultLoginValve();
115        pipeline.addValve(valve);
116        pipeline.initialize();
117
118        pipeline.invoke(pipelineData);
119        User user = runData.getUser();
120        assertNotNull(user);
121        assertEquals("username",user.getName());
122        assertTrue(user.hasLoggedIn());
123    }
124
125    /**
126     * Tests the LogoutAction.
127     */
128    @Test
129    public void testLogout() throws Exception
130    {
131        User user = securityService.getUser("username");
132        user.setHasLoggedIn(Boolean.TRUE);
133        session.setAttribute(User.SESSION_KEY, user);
134
135        RunData runData = getRunData(request,response,config);
136        runData.setAction(TurbineConstants.ACTION_LOGOUT_DEFAULT);
137
138        Pipeline pipeline = new TurbinePipeline();
139        PipelineData pipelineData = runData;
140
141        DefaultLoginValve valve = new DefaultLoginValve();
142        pipeline.addValve(valve);
143        pipeline.initialize();
144
145        pipeline.invoke(pipelineData);
146        user = runData.getUser();
147        assertNotNull(user);
148        assertTrue(securityService.isAnonymousUser(user));
149        assertTrue(securityService.isAnonymousUser( (User)session.getAttribute( User.SESSION_KEY )));
150        assertFalse(user.hasLoggedIn());
151    }
152
153    @AfterAll
154    public static void destroy()
155    {
156        tc.dispose();
157    }
158}
