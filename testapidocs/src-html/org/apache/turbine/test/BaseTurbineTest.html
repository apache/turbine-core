022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.turbine.TurbineConstants;
026import org.apache.turbine.util.TurbineConfig;
027import org.junit.BeforeClass;
028
029/**
030 * A base class to implement tests that need a running
031 * Turbine framework on it.
032 *
033 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
034 * @version $Id: BaseTurbineTest.java 534527 2007-05-02 16:10:59Z tv $
035 */
036
037public abstract class BaseTurbineTest
038        extends BaseTestCase
039{
040    private static volatile TurbineConfig turbineConfig = null;
041
042    @BeforeClass
043    public static void turbineInit(String config) throws Exception
044    {
045        if (turbineConfig == null)
046        {
047            Map<String, String> initParams = new HashMap<>();
048            initParams.put(TurbineConfig.PROPERTIES_PATH_KEY, config); // "conf/test/TurbineResources.properties"
049            initParams.put(TurbineConstants.LOGGING_ROOT_KEY, "target/test-logs");
050
051            TurbineConfig _turbineConfig = new TurbineConfig(".", initParams);
052            _turbineConfig.initialize();
053
054            turbineConfig = _turbineConfig;
055        }
056    }
057}
