024import static org.junit.Assert.assertTrue;
025
026import org.apache.turbine.test.BaseTestCase;
027import org.junit.Test;
028
029public class FormMessageTest
030    extends BaseTestCase
031{
032
033
034    @Test public void testCreateFormMessage()
035    {
036        FormMessage fm = new FormMessage("mainForm","someField","A message!");
037        String fmString = fm.toString();
038        assertTrue("Make sure toString works",fmString.indexOf("someField")>-1);
039        assertTrue("Make sure toString works",fmString.indexOf("A message")>-1);
040        assertTrue("Make sure toString works",fmString.indexOf("mainForm")>-1);
041    }
042
043    @Test public void testCreateFormMessageMultipleFields()
044    {
045        FormMessage fm = new FormMessage("mainForm","someField","A message!");
046        fm.setFieldName("someOtherField");
047        String fmString = fm.toString();
048        assertTrue("Make sure toString works",fmString.indexOf("someField")>-1);
049        assertTrue("Make sure toString works",fmString.indexOf("someOtherField")>-1);
050        assertTrue("Make sure toString works",fmString.indexOf("A message")>-1);
051        assertTrue("Make sure toString works",fmString.indexOf("mainForm")>-1);
052    }
053}
