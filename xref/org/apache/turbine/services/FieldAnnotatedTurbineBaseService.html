25  import org.apache.turbine.annotation.AnnotationProcessor;
26  import org.apache.turbine.util.TurbineException;
27  
28  /**
29   * <p>This class provides a <code>Service</code> implementation that
30   * Services used in Turbine are required to extend. 
31   * This class provides the ability to process field annotation {@link TurbineServices} in a Turbine service.
32   * You could enable scanning globally by annotating the class (service) with the annotation {@link TurbineServices}.
33   * Field annotation could then be omitted, if the field class is {@link TurbineServices} annotated.
34   * </p>
35   *
36   */
37  public abstract class FieldAnnotatedTurbineBaseService
38          extends TurbineBaseService
39  {
40      
41      private static Logger log = LogManager.getLogger(FieldAnnotatedTurbineBaseService.class);
42      /**
43       * Performs late initialization.
44       *
45       * If your class relies on early initialization, and the object it
46       * expects was not received, you can use late initialization to
47       * throw an exception and complain.
48       *
49       * @throws InitializationException if initialization of this
50       * class was not successful.
51       */
52      @Override
53      public void init() throws InitializationException
54      {
55          try {
56              log.debug("parsing annotations for {}", this.getClass());
57              AnnotationProcessor.process(this, false);
58          } catch (TurbineException e) {
59              throw new InitializationException(e.getMessage(), e);
60          }
61          setInit(true);
62      }
63  }
