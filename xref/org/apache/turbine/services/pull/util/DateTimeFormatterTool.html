41  /**
42   * This pull tool is used to format {@link TemporalAccessor} and
43   * {@link #map(String, DateTimeFormatter, Locale)} (different flavors)
44   * objects into strings.
45   * 
46   * This tool extends {@link DateFormatter} to simplify configuration 
47   * and to allow legacy {@link Date} inputs.
48   *
49   * The methods may throw {@link java.time.temporal.UnsupportedTemporalTypeException} or
50   * {@link DateTimeParseException}.
51   * if the source and the target format do not match appropriately.
52   *
53   */
54  public class DateTimeFormatterTool extends DateFormatter
55          implements DateTimeFormatterInterface
56  {
57  
58      @TurbineService
59      private DateTimeFormatterService dtfs;
60  
61      private static final Logger log = LogManager.getLogger(DateTimeFormatterTool.class);
62  
63      /** Fulcrum Localization component */
64      @TurbineService
65      private LocalizationService localizationService;
66      
67      protected Locale locale;
68      
69      private boolean overrideFromRequestLocale = false;
70      
71      /**
72       * Initialize the application tool. The data parameter holds a different
73       * type depending on how the tool is being instantiated:
74       * <ul>
75       * <li>For global tools data will be null</li>
76       * <li>For request tools data will be of type RunData</li>
77       * <li>For session and persistent tools data will be of type User</li>
78       * </ul>
79       *
80       * the {@link #getDefaultFormat()} from {@link #getFormatPattern()}
81       * with {@link DateTimeFormatterService#getLocale()}
82       * and zoneId {@link DateTimeFormatterService#getZoneId()} is used.
83       * 
84       * Customizations:
85       * Locale could be fetched from request, if #USE_REQUEST_LOCALE_KEY is set to 
86       * <code>true</code> (by default it is <code>false</code>.Then it will be retrieved either from 
87       * {@link RundataLocalizationService#getLocale(RunData)} (if set in urbien role configuration)
88       * or {@link LocalizationService#getLocale(javax.servlet.http.HttpServletRequest)}.
89       * 
90       * @param data initialization data
91       */
92      @Override
93      public void init(Object data)
94      {
95          log.info("Initializing DateTimeFormatterTool with service {}",
96                  dtfs);
97          if (dtfs == null)
98          {
99              ServiceManager serviceManager = TurbineServices.getInstance();
100             dtfs = (DateTimeFormatterService)serviceManager.getService(DateTimeFormatterService.SERVICE_NAME);
101         }
102         
103         overrideFromRequestLocale = Turbine.getConfiguration()
104                 .getBoolean(USE_REQUEST_LOCALE_KEY, false);
105         // dtfs should be already initialized
106         if (overrideFromRequestLocale && data instanceof RunData)
107         {
108             // Pull necessary information out of RunData while we have
109             // a reference to it.
110             locale = localizationService.getLocale(((RunData) data).getRequest());
111             log.info("Override {} with request locale {} from {}", dtfs.getLocale(), locale, localizationService);
112         }
113     }
114 
115     public DateTimeFormatterService getDtfs() {
116         return dtfs;
117     }
118 
119     /**
120      * Refresh the application tool. This is
121      * necessary for development work where you
122      * probably want the tool to refresh itself
123      * if it is using configuration information
124      * that is typically cached after initialization
125      */
126     @Override
127     public void refresh()
128     {
129         // empty
130     }
131 
132     @Override
133     public DateTimeFormatter getDefaultFormat()
134     {
135         return getDtfs().getDefaultFormat();
136     }
137 
138     @Override
139     public String getFormatPattern() {
140         return getDtfs().getFormatPattern();
141     }
142 
143     /**
144      * Formats the given datetime as a String with the #{@link DateTimeFormatterTool#getDefaultFormat()}.
145      * using the default date format.
146      *
147      * @param temporalAccessor {@link TemporalAccessor to format
148      * @return String value of the date
149      */
150     @Override
151     public <T extends TemporalAccessor> String format(T temporalAccessor)
152     {
153         return getDtfs().format(temporalAccessor, getDtfs().getFormatPattern(), getLocale());
154     }
155 
156     @Override
157     public <T extends TemporalAccessor> String format(T temporalAccessor, String dateFormatString)
158     {
159         return getDtfs().format(temporalAccessor, dateFormatString, getLocale());
160     }
161 
162     @Override
163     public <T extends TemporalAccessor> String format(T temporalAccessor, String dateFormatString, Locale locale)
164     {
165         return getDtfs().format(temporalAccessor, dateFormatString, locale);
166     }
167 
168     @Override
169     public <T extends TemporalAccessor> String format(T temporalAccessor, String dateFormatString, Locale locale,
170             ZoneId zoneId) {
171         return getDtfs().format(temporalAccessor, dateFormatString, locale, zoneId);
172     }
173 
174     @Override
175     public String map( String src, String outgoingFormatPattern, Locale locale, String incomingFormatPattern)
176     {
177         return getDtfs().map(src, outgoingFormatPattern, locale, incomingFormatPattern);
178     }
179 
180     @Override
181     public String map( String src, java.time.format.DateTimeFormatter outgoingFormat, 
182             Locale locale, java.time.format.DateTimeFormatter incomingFormat)
183     {
184         return getDtfs().map(src, outgoingFormat, locale, incomingFormat);
185     }
186 
187     @Override
188     public String mapTo( String src, DateTimeFormatter outgoingFormat )
189     {
190         return  getDtfs().map( src, outgoingFormat, getLocale(), getDtfs().getDefaultFormat() );
191     }
192 
193     @Override
194     public String mapFrom( String src, DateTimeFormatter incomingFormat )
195     {
196         return  getDtfs().map( src, getDtfs().getDefaultFormat(), getLocale(), incomingFormat );
197     }
198 
199     @Override
200     public String map( String src,  DateTimeFormatter outgoingFormat, Locale locale )
201     {
202         return  getDtfs().map( src, outgoingFormat, locale, getDtfs().getDefaultFormat() );
203     }
204 
205     public Locale getLocale() {
206         return locale;
207     }
208 
209     public void setLocale(Locale locale) {
210         this.locale = locale;
211     }
212 
213     @Override
214     public ZoneId getZoneId() {
215         return (getDtfs()!= null)? getDtfs().getZoneId():ZoneId.systemDefault();
216     }
217 
218 }
