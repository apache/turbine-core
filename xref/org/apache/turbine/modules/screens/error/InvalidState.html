32  /**
33   * Users will get this screen if the screen on their browser is in an
34   * invalid state.  For example, if they hit "Back" or "Reload" and
35   * then try to submit old form data.
36   *
37   * If you want one of your screens to check for invalid state
38   * then add a hidden form field called "_session_access_counter"
39   * with the value currently stored in the session.  The
40   * SessionValidator action will check to see if it is an old
41   * value and redirect you to this screen.
42   *
43   * @author <a href="mailto:mbryson@mont.mindspring.com">Dave Bryson</a>
44   * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
45   * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux</a>
46   * @version $Id: InvalidState.java 1709648 2015-10-20 17:08:10Z tv $
47   */
48  public class InvalidState
49      extends Screen
50  {
51      /**
52       * Build the Screen.
53       *
54       * @param pipelineData Turbine information.
55       * @exception Exception, a generic exception.
56       */
57      @Override
58      public String doBuild(PipelineData pipelineData)
59              throws Exception
60      {
61          RunData data = getRunData(pipelineData);
62          ElementContainer body = new ElementContainer();
63          ElementContainer message = new ElementContainer();
64  
65          StringBuilder sb = new StringBuilder();
66          sb.append("<b>There has been an error.</b>")
67                  .append("<p>")
68                  .append("- If you used the browser \"Back\" or \"Reload\"")
69                  .append(" buttons please use the navigation buttons we provide")
70                  .append(" within the screen.")
71                  .append("<p>")
72                  .append("Please click ");
73  
74          message.addElement(sb.toString());
75          ParameterParser pp;
76          pp = (ParameterParser) data.getUser().getTemp("prev_parameters");
77          pp.remove("_session_access_counter");
78  
79          TurbineURI back = new TurbineURI(data,(String) data.getUser().getTemp("prev_screen"));
80          back.addPathInfo(pp);
81          message.addElement(new A().setHref(back.getRelativeLink()).addElement("here"));
82  
83          message.addElement(" to return the the screen you were working on.");
84  
85          body.addElement(message);
86          return body.toString();
87      }
88  }
