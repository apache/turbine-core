59      private static ConcurrentMap<String, Annotation[]> annotationCache = new ConcurrentHashMap<>();

61      /**
62       * Get cached annotations for field, class or method
63       *
64       * @param object a field, class or method
65       *
66       * @return the declared annotations for the object
67       */
68      public static Annotation[] getAnnotations(AccessibleObject object)
69      {
70          String key = object.getClass() + object.toString();
71          Annotation[] annotations = annotationCache.get(key);
72          if (annotations == null)
73          {
74              Annotation[] newAnnotations = object.getDeclaredAnnotations();
75              annotations = annotationCache.putIfAbsent(key, newAnnotations);
76              if (annotations == null)
77              {
78                  annotations = newAnnotations;
79              }
80          }
81          return annotations;
82      }

84      public enum ConditionType
85      {
86          COMPOUND, ANY;

89      /**
90       * Check if the object given is authorized to be executed based on its annotations
91       *
92       * The method will return false if one of the annotations denies execution
93       *
94       * @see #isAuthorized(AccessibleObject, TurbineAccessControlList, ConditionType)
95       *
96       * @param <A> ACL instance
97       * @param object accessible object to test
98       * @param acl access control list
99       * @return true if the execution is allowed
100      */
101     public static <A extends TurbineAccessControlList<?>> boolean isAuthorized(AccessibleObject object, A acl)
102     {
103         return isAuthorized( object, acl, ConditionType.COMPOUND );
104     }

106     /**
107      * Check if the object given is authorized to be executed based on its annotations
108      * The method's return value depends on the conditonType, refer to the ConditionType
109      *
110      * @param <A> ACL instance
111      * @param object the object
112      * @param acl access control list
113      * @param conditonType either {@link ConditionType#COMPOUND}: The method will return false if one of the annotations denies execution
114      *                     or {@link ConditionType#ANY} : The method will return true if one of the annotations allows execution
115      * @return true if the execution is allowed
116      */
117     public static <A extends TurbineAccessControlList<?>> boolean isAuthorized(AccessibleObject object, A acl, ConditionType conditonType)
118     {
119         Annotation[] annotations = getAnnotations(object);
120 
121         for (Annotation annotation : annotations)
122         {
123             if (annotation instanceof TurbineRequiredRole)
124             {
125                 TurbineRequiredRole trr = (TurbineRequiredRole) annotation;
126                 String[] roleNames = trr.value();
127                 String group = trr.group();
128 
129                 if (StringUtils.isEmpty(group)) // global group
130                 {
131                     for (String roleName : roleNames)
132                     {
133                         switch ( conditonType ) {
134                             case COMPOUND: default:
135                                 if (!acl.hasRole(roleName))
136                                 {
137                                     return false;
138                                 }
139                                 break;
140                             case ANY:
141                                 if (acl.hasRole(roleName))
142                                 {
143                                     return true;
144                                 }
145                                 break;
146                         }
147                     }
148                     if (conditonType == ConditionType.ANY) { // nothing matched
149                         return false;
150                     }
151                 }
152                 else
153                 {
154                     for (String roleName : roleNames)
155                     {
156                         switch ( conditonType ) {
157                             case COMPOUND: default:
158                                 if (!acl.hasRole(roleName, group))
159                                 {
160                                     return false;
161                                 }
162                                 break;
163                             case ANY:
164                                 if (acl.hasRole(roleName, group))
165                                 {
166                                     return true;
167                                 }
168                                 break;
169                         }
170                     }
171                 }
172             }
173             else if (annotation instanceof TurbineRequiredPermission)
174             {
175                 TurbineRequiredPermission trp = (TurbineRequiredPermission) annotation;
176                 String[] permissionNames = trp.value();
177                 String group = trp.group();
178 
179                 if (StringUtils.isEmpty(group)) // global group
180                 {
181                     for (String permissionName : permissionNames)
182                     {
183                         switch ( conditonType ) {
184                             case COMPOUND: default:
185                                 if (!acl.hasPermission(permissionName))
186                                 {
187                                     return false;
188                                 }
189                                 break;
190                             case ANY:
191                                 if (acl.hasPermission(permissionName))
192                                 {
193                                     return true;
194                                 }
195                                 break;
196                         }
197                     }
198                 }
199                 else
200                 {
201                     for (String permissionName : permissionNames)
202                     {
203                         switch ( conditonType ) {
204                             case COMPOUND: default:
205                                 if (!acl.hasPermission(permissionName, group))
206                                 {
207                                     return false;
208                                 }
209                                 break;
210                             case ANY:
211                                 if (acl.hasPermission(permissionName, group))
212                                 {
213                                     return true;
214                                 }
215                                 break;
216                         }
217 
218                     }
219                 }
220             }
221         }
222 
223         return true;
224     }

226     /**
227      * Search for annotated fields of the object and inject the appropriate
228      * objects
229      *
230      * @param object the object
231      * @throws TurbineException if the objects could not be injected
232      */
233     public static void process(Object object) throws TurbineException
234     {
235         process(object, false);
236     }

238     /**
239      * Search for annotated fields and optionally of method fields of the object and inject the appropriate
240      * objects
241      *
242      * @param object the object
243      * @param hasTurbineServicesInMethodFields set <code>true </code>, if methods should be parsed
244      * @throws TurbineException if the objects could not be injected
245      */
246     public static void process(Object object, Boolean hasTurbineServicesInMethodFields) throws TurbineException
247     {
248         ServiceManager manager = null;
249         Configuration config = null;
250         AssemblerBrokerService assembler = null;
251         PoolService pool= null;
252         Class<?> clazz = object.getClass();
253         
254         boolean isTurbineService = false;
255         if ( clazz.isAnnotationPresent(TurbineService.class)) {
256             TurbineService service = clazz.getAnnotation(TurbineService.class);
257             log.debug("retrieved class annotation: "+ service);
258             isTurbineService = true;
259         } 
260 
261         while (clazz != null)
262         {
263             Field[] fields = clazz.getDeclaredFields();
264 
265             for (Field field : fields)
266             {
267                 Annotation[] annotations = getAnnotations(field);
268 
269                 for (Annotation a : annotations)
270                 {
271                     if (a instanceof TurbineService)
272                     {
273                         if (manager == null)
274                         {
275                             manager = TurbineServices.getInstance();
276                         }
277                         injectTurbineService(object, manager, field, (TurbineService) a);
278                     }
279                     else if (a instanceof TurbineConfiguration)
280                     {
281                         if (config == null)
282                         {
283                             config = Turbine.getConfiguration();
284                         }
285                         injectTurbineConfiguration(object, config, field, (TurbineConfiguration) a);
286                     }
287                     else if (a instanceof TurbineLoader)
288                     {
289                         if (assembler == null)
290                         {
291                             assembler = (AssemblerBrokerService) TurbineServices.getInstance().
292                                 getService(AssemblerBrokerService.SERVICE_NAME);
293                         }
294                         injectTurbineLoader(object, assembler, field, (TurbineLoader) a);
295                     }
296                     else if (a instanceof TurbineTool)
297                     {
298                         if (pool == null)
299                         {
300                             pool = (PoolService)TurbineServices.getInstance()
301                                     .getService(PoolService.ROLE);
302                         }
303                         injectTurbineTool(object, pool, field, (TurbineTool) a);
304                     }
305                 }
306                 if (isTurbineService)
307                 {
308                     if (field.getType().isAnnotationPresent(TurbineService.class)) {
309                         TurbineService service = field.getType().getAnnotation(TurbineService.class);
310                         log.debug("retrieved implicit class annotation: "+ service);
311                         if (manager == null)
312                         {
313                             manager = TurbineServices.getInstance();
314                         }
315                         injectTurbineService(object, manager, field, service);
316                     }    
317                 }
318             }
319 
320             if (hasTurbineServicesInMethodFields) {
321                 manager = processMethods(object, manager, clazz, isTurbineService);
322             }
323 
324             clazz = clazz.getSuperclass();
325         }
326     }
327 
328     private static ServiceManager processMethods(Object object, ServiceManager manager, Class<?> clazz, boolean isTurbineService) throws TurbineException {
329         Method[] methods = clazz.getMethods();
330 
331         for (Method method : methods)
332         {
333             Annotation[] annotations = getAnnotations(method);
334             for (Annotation a : annotations)
335             {
336                 if (a instanceof TurbineService)
337                 {
338 
339                     if (manager == null)
340                     {
341                         manager = TurbineServices.getInstance();
342                     }
343                     injectTurbineService(object, manager, method, (TurbineService) a);
344                 }
345             }
346             if (isTurbineService)
347             {
348                 if (manager == null)
349                 {
350                     manager = TurbineServices.getInstance();
351                 }
352                 Class<?>[] classes = method.getParameterTypes();
353                 for (Class<?> c : classes)
354                 {
355                     if ( c.isAnnotationPresent(TurbineService.class)) {
356                         TurbineService service = c.getAnnotation(TurbineService.class);
357                         log.debug("retrieved implicit service in Turbien service: "+ service);
358                         injectTurbineService(object, manager, method, service);
359                     } 
360                     
361                 }
362             }
363         }
364         return manager;
365     }

367     /**
368      * Inject Turbine loader into field of object
369      *
370      * @param object the object to process
371      * @param assembler AssemblerBrokerService, provides the loader
372      * @param field the field
373      * @param annotation the value of the annotation
374      *
375      * @throws TurbineException if loader cannot be set
376      */
377     private static void injectTurbineLoader(Object object, AssemblerBrokerService assembler, Field field, TurbineLoader annotation) throws TurbineException
378     {
379         Loader<?> loader = assembler.getLoader(annotation.value());
380         field.setAccessible(true);
381 
382         try
383         {
384             log.debug("Injection of {} into object {}", loader, object);
385 
386             field.set(object, loader);
387         }
388         catch (IllegalArgumentException | IllegalAccessException e)
389         {
390             throw new TurbineException("Could not inject loader "
391                     + loader + " into object " + object, e);
392         }
393     }

395     /**
396      * Inject Turbine tool into field of object and 
397      * injects annotations provided in the tool.
398      *
399      * @param object the object to process
400      * @param pool PoolService, provides the pool
401      * @param field the field
402      * @param annotation the value of the annotation
403      *
404      * @throws TurbineException if loader cannot be set
405      */
406     private static void injectTurbineTool(Object object, PoolService pool, Field field, TurbineTool annotation) throws TurbineException
407     {
408         Object tool = null;
409         try
410         {
411             tool = pool.getInstance(annotation.value());
412             // inject annotations in tool
413             process(tool);
414 
415             field.setAccessible(true);
416             log.debug("Injection of {} into object {}", tool, object);
417 
418             field.set(object, tool);
419         }
420         catch (PoolException | IllegalArgumentException | IllegalAccessException e)
421         {
422             throw new TurbineException("Could not inject tool "
423                     + tool + " into object " + object, e);
424         } 
425     }

427     /**
428      * Inject Turbine configuration into field of object
429      *
430      * @param object the object to process
431      * @param conf the configuration to use
432      * @param field the field
433      * @param annotation the value of the annotation
434      *
435      * @throws TurbineException if configuration cannot be set
436      */
437     @SuppressWarnings("boxing")
438     private static void injectTurbineConfiguration(Object object, Configuration conf, Field field, TurbineConfiguration annotation) throws TurbineException
439     {
440         Class<?> type = field.getType();
441         String key = annotation.value();
442 
443         try
444         {
445             if (Configuration.class.isAssignableFrom(type))
446             {
447                 final Configuration injectConfiguration;
448                 // Check for annotation value
449                 if (StringUtils.isNotEmpty(key))
450                 {
451                     injectConfiguration = conf.subset(key);
452                 }
453                 else
454                 {
455                     injectConfiguration = conf;
456                 }
457 
458                 log.debug("Injection of {} into object {}", injectConfiguration, object);
459 
460                 field.setAccessible(true);
461                 field.set(object, injectConfiguration);
462             }
463             else if (conf.containsKey(key))
464             {
465                 if ( String.class.isAssignableFrom( type ) )
466                 {
467                     String value = conf.getString(key);
468                     log.debug("Injection of key {} into object {}", value, object);
469 
470                     field.setAccessible(true);
471                     field.set(object, value);
472                 }
473                 else if ( Boolean.TYPE.isAssignableFrom( type ) )
474                 {
475                     boolean value = conf.getBoolean(key);
476                     log.debug("Injection of key {} into object {}", value, object);
477 
478                     field.setAccessible(true);
479                     field.setBoolean(object, value);
480                 }
481                 else if ( Integer.TYPE.isAssignableFrom( type ) )
482                 {
483                     int value = conf.getInt(key);
484                     log.debug("Injection of key {} into object {}", value, object);
485 
486                     field.setAccessible(true);
487                     field.setInt(object, value);
488                 }
489                 else if ( Long.TYPE.isAssignableFrom( type ) )
490                 {
491                     long value = conf.getLong(key);
492                     log.debug("Injection of key {} into object {}", value, object);
493 
494                     field.setAccessible(true);
495                     field.setLong(object, value);
496                 }
497                 else if ( Short.TYPE.isAssignableFrom( type ) )
498                 {
499                     short value = conf.getShort(key);
500                     log.debug("Injection of key {} into object {}", value, object);
501 
502                     field.setAccessible(true);
503                     field.setShort(object, value);
504                 }
505                 else if ( Long.TYPE.isAssignableFrom( type ) )
506                 {
507                     long value = conf.getLong(key);
508                     log.debug("Injection of key {} into object {}", value, object);
509 
510                     field.setAccessible(true);
511                     field.setLong(object, value);
512                 }
513                 else if ( Float.TYPE.isAssignableFrom( type ) )
514                 {
515                     float value = conf.getFloat(key);
516                     log.debug("Injection of key {} into object {}", value, object);
517 
518                     field.setAccessible(true);
519                     field.setFloat(object, value);
520                 }
521                 else if ( Double.TYPE.isAssignableFrom( type ) )
522                 {
523                     double value = conf.getDouble(key);
524                     log.debug("Injection of key {} into object {}", value, object);
525 
526                     field.setAccessible(true);
527                     field.setDouble(object, value);
528                 }
529                 else if ( Byte.TYPE.isAssignableFrom( type ) )
530                 {
531                     byte value = conf.getByte(key);
532                     log.debug("Injection of key {} into object {}", value, object);
533 
534                     field.setAccessible(true);
535                     field.setByte(object, value);
536                 }
537                 else if ( List.class.isAssignableFrom( type ) )
538                 {
539                     List<Object> values = conf.getList(key);
540                     log.debug("Injection of key {} into object {}", values, object);
541 
542                     field.setAccessible(true);
543                     field.set(object, values);
544                 } else {
545                     throw new TurbineException("Could not inject type " + 
546                       type + " into object " + object + ". Type "+ type + " not assignable in configuration "
547                       + conf + " (allowed: String, Boolean, List, Number Types, "+ Configuration.class.getName() + ").");
548                 }
549             } else {
550                 field.setAccessible(true);
551                 Object defaultValue = field.get(object);
552                 // this should not throw an error as it might be set later from container  e. g. session.timeout 
553                 // we might check field.get<Type> to show the default value of the field, but this is only a guess, it might be set even later..
554                 log.info("No key {} of type {} injected into object {}. Field {} is set to default {}.", key, type, object, field.getName(), defaultValue);
555             }
556         }
557         catch (IllegalArgumentException | IllegalAccessException e)
558         {
559             throw new TurbineException("Could not inject configuration "
560                     + conf + " into object " + object, e);
561         }
562     }

564     /**
565      * Inject Turbine service into field of object
566      *
567      * @param object the object to process
568      * @param manager the service manager
569      * @param field the field
570      * @param annotation the value of the annotation
571      *
572      * @throws TurbineException if service is not available
573      */
574     private static void injectTurbineService(Object object, ServiceManager manager, Field field, TurbineService annotation) throws TurbineException
575     {
576         String serviceName = null;
577         // Check for annotation value
578         if (annotation != null && StringUtils.isNotEmpty(annotation.value()))
579         {
580             serviceName = annotation.value();
581         }
582         // Check for fields SERVICE_NAME and ROLE
583         else
584         { 
585             // check field level annotation
586             Field[] typeFields = field.getType().getFields();
587             serviceName = checkServiceOrRoleInField(serviceName, typeFields);
588             // if it is the default Service, we check class level annotation
589             if ( (serviceName == null || serviceName.equals(Service.SERVICE_NAME)) &&
590                     field.getType().isAnnotationPresent(TurbineService.class)) {
591                 TurbineService service = field.getType().getAnnotation(TurbineService.class);
592                 log.debug("retrieved class annotation: "+ service);
593                 serviceName = service.value();
594             } 
595         }
596 
597         if (StringUtils.isEmpty(serviceName))
598         {
599             // Try interface class name (e.g. used by Fulcrum)
600             serviceName = field.getType().getName();
601         }
602 
603         log.debug("Looking up service for injection: {} for object {}", serviceName, object);
604 
605         Object service = manager.getService(serviceName); // throws Exception on unknown service
606         field.setAccessible(true);
607 
608         try
609         {
610             log.debug("Injection of {} into object {}", serviceName, object);
611 
612             field.set(object, service);
613         }
614         catch (IllegalArgumentException | IllegalAccessException e)
615         {
616             throw new TurbineException("Could not inject service "
617                     + serviceName + " into object " + object, e);
618         }
619     }

621     /**
622      * Injects Turbine service into method fields 
623      * 
624      * @param object the object to process
625      * @param manager the service manager
626      * @param method The method
627      * @param annotation the value of the annotation
628      * @throws TurbineException - If service could not be injected.
629      */
630     private static void injectTurbineService(Object object, ServiceManager manager, Method method, TurbineService annotation) throws TurbineException
631     {
632         String serviceName = null;
633         // Check for annotation value
634         if (annotation != null && StringUtils.isNotEmpty(annotation.value()))
635         {
636             serviceName = annotation.value();
637         }
638         else
639         {
640             Class<?>[] classes = method.getParameterTypes();
641             for (Class<?> c : classes)
642             {
643                 Field[] fields = c.getFields();
644                 // Check for fields SERVICE_NAME and ROLE
645                 serviceName = checkServiceOrRoleInField(serviceName, fields);
646                 
647                 if ( (serviceName == null || serviceName.equals(Service.SERVICE_NAME)) &&
648                         c.isAnnotationPresent(TurbineService.class)) {
649                     TurbineService service = c.getAnnotation(TurbineService.class);
650                     log.debug("retrieved class annotation: "+ service);
651                     serviceName = service.value();
652                 } 
653                 
654             }
655         }
656 
657         log.debug("Looking up service for injection: {} for object {}", serviceName, object);
658         if (StringUtils.isEmpty(serviceName))
659         {
660             // Try interface class name
661             serviceName = method.getName();
662         }
663 
664         Object service = manager.getService(serviceName); // throws Exception on unknown service
665         method.setAccessible(true);
666 
667         try
668         {
669             log.debug("Injection of {} into object {}", serviceName, object);
670 
671             Object[] paramValues = new Object[1];
672             paramValues[0] = service;
673             method.invoke(object, paramValues);
674         }
675         catch (IllegalArgumentException | IllegalAccessException | InvocationTargetException e)
676         {
677             throw new TurbineException("Could not inject service "
678                     + serviceName + " into object " + object, e);
679         }
680     }

682     private static String checkServiceOrRoleInField(String serviceName, Field[] fields) {
683         for (Field f : fields)
684             if (TurbineService.SERVICE_NAME.equals(f.getName()))
685             {
686                 try
687                 {
688                     serviceName = (String)f.get(null);
689                 }
690                 catch (IllegalArgumentException | IllegalAccessException e)
691                 {
692                     continue;
693                 }
694                 break;
695             }
696             else if (TurbineService.ROLE.equals(f.getName()))
697             {
698                 try
699                 {
700                     serviceName = (String)f.get(null);
701                 }
702                 catch (IllegalArgumentException | IllegalAccessException e)
703                 {
704                     continue;
705                 }
706                 break;
707             }
708         return serviceName;
709     }
