22  import static org.junit.jupiter.api.Assertions.assertTrue;
23  import static org.junit.jupiter.api.Assertions.assertEquals;

26  import java.io.StringReader;
27  import java.io.StringWriter;
28  
29  import javax.xml.bind.JAXBContext;
30  import javax.xml.bind.Marshaller;
31  import javax.xml.bind.Unmarshaller;
32  
33  import org.junit.jupiter.api.BeforeEach;
34  
35  import org.junit.jupiter.api.Test;
36  
37  /**
38   * Tests TurbinePipeline.
39   *
40   * @author <a href="mailto:dlr@finemaltcoding.com">Daniel Rall</a>
41   * @version $Id: PipelineCreationTest.java 1853146 2019-02-07 16:14:46Z gk $
42   */
43  public class PipelineCreationTest
44  {
45      private Pipeline pipeline;
46  
47      @BeforeEach
48      public void setUp()
49      {
50          pipeline = new TurbinePipeline();
51          pipeline.addValve(new SimpleValve());
52          pipeline.addValve(new DetermineActionValve());
53      }
54  
55      @Test
56      public void testSavingPipeline() throws Exception
57      {
58          JAXBContext context = JAXBContext.newInstance(TurbinePipeline.class);
59          Marshaller marshaller = context.createMarshaller();
60          StringWriter writer = new StringWriter();
61          marshaller.marshal(pipeline, writer);
62          assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
63                  + "<pipeline><valves>"
64                  + "<valve>org.apache.turbine.pipeline.SimpleValve</valve>"
65                  + "<valve>org.apache.turbine.pipeline.DetermineActionValve</valve>"
66                  + "</valves></pipeline>", writer.toString());
67      }
68  
69      @Test
70      public void testReadingPipeline() throws Exception
71      {
72          String xml = "<pipeline name=\"default\"><valves>"
73                  + "<valve>org.apache.turbine.pipeline.SimpleValve</valve>"
74                  + "<valve>org.apache.turbine.pipeline.DetermineActionValve</valve>"
75                  + "</valves></pipeline>";
76          JAXBContext context = JAXBContext.newInstance(TurbinePipeline.class);
77          Unmarshaller unmarshaller = context.createUnmarshaller();
78          StringReader reader = new StringReader(xml);
79          Pipeline pipeline = (Pipeline) unmarshaller.unmarshal(reader);
80          assertEquals(2, pipeline.getValves().length);
81          assertTrue(pipeline.getValves()[0] instanceof SimpleValve);
82          assertTrue(pipeline.getValves()[1] instanceof DetermineActionValve);
83      }
84  
85  }
