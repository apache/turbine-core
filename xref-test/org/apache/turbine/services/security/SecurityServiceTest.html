106     /**
107      * Tests Turbine and Fulcrum.
108      *
109      * Fulcrum part is similar/duplicated from {@link AbstractTurbineModelManagerTest#testGrantUserGroupRole()}
110      *
111      *
112      * @throws Exception
113      */
114     @Test
115     public void testGrantUserGroupRole() throws Exception
116     {
117         Group group = fulcrumSecurityService.getGroupManager().getGroupInstance();
118         group.setName("TEST_GROUP");
119         fulcrumSecurityService.getGroupManager().addGroup(group);
120         Role role = fulcrumSecurityService.getRoleManager().getRoleInstance();
121         role.setName("TEST_Role");
122         fulcrumSecurityService.getRoleManager().addRole(role);
123 
124         //  Turbine security service returns a wrapped instance: org.apache.turbine.om.security.DefaultUserImpl
125         // which implements org.apache.turbine.om.security.User and contains
126         User user = securityService.getUserInstance("Clint");
127 		// memory
128         securityService.addUser(user, "clint");
129         securityService.grant(user, group, role);
130 
131 		addUserAndCheck(group, role, user.getUserDelegate());
132 
133         // Fulcrum security service returns a raw org.apache.fulcrum.security.model.turbine.entity.impl.TurbineUserImpl,
134 		org.apache.fulcrum.security.UserManager  userManager = fulcrumSecurityService.getUserManager();
135 		TurbineUser fulcrumUser = userManager.getUserInstance("Clint2");
136         userManager.addUser(fulcrumUser, "clint2");         // memory
137         ((TurbineModelManager)fulcrumSecurityService.getModelManager()).grant(fulcrumUser, group, role);
138 
139         addUserAndCheck(group, role, fulcrumUser);
140 
141     }
142 
143     /**
144      * Fulcrum contract check
145      *
146      * @param group Fulcrum interface
147      * @param role Fulcrum interface
148      * @param user Fulcrum interface
149      * @throws EntityExistsException
150      * @throws DataBackendException
151      * @throws UnknownEntityException
152      */
153 	private void addUserAndCheck(Group group, Role role, TurbineUser user)
154 			throws EntityExistsException, DataBackendException,
155 			UnknownEntityException
156 	{
157 
158         boolean ugrFound = false;
159         TurbineUserGroupRole ugrTest = null;
160         for (TurbineUserGroupRole ugr : user.getUserGroupRoleSet())
161         {
162             if (ugr.getUser().equals(user) && ugr.getGroup().equals(group) && ugr.getRole().equals(role))
163             {
164                 ugrFound = true;
165                 ugrTest = ugr;
166                 break;
167             }
168         }
169         assertTrue(ugrFound);
170         assertTrue(ugrTest.getGroup().equals(group));
171         assertTrue(ugrTest.getUser().equals(user));
172 	}
