22  import static org.junit.Assert.assertEquals;
23  
24  import org.apache.turbine.TurbineConstants;
25  import org.junit.Test;
26  
27  /**
28   * Testing of the HtmlPageAttributes class
29   *
30   * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
31   */
32  public class HtmlPageAttributesTest
33  {
34      @Test public void testBuildDoctype()
35      {
36          HtmlPageAttributes page = new HtmlPageAttributes();
37          assertEquals("<!DOCTYPE html>", page.getDoctype("html", null, null));
38          assertEquals("<!DOCTYPE html>", page.getDoctype("html", "", ""));
39          assertEquals("<!DOCTYPE html SYSTEM \"bla\">", page.getDoctype("html", "", "bla"));
40  
41          assertEquals("<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01 Transitional//EN\" "
42                  + "\"http://www.w3.org/TR/1999/REC-html401-19991224/loose.dtd\">",
43                  page.getDoctype(TurbineConstants.DEFAULT_HTML_DOCTYPE_ROOT_ELEMENT_DEFAULT,
44                          TurbineConstants.DEFAULT_HTML_DOCTYPE_IDENTIFIER_DEFAULT,
45                          TurbineConstants.DEFAULT_HTML_DOCTYPE_URI_DEFAULT));
46  
47          // HTML 5 is the default
48          assertEquals("<!DOCTYPE HTML>", page.getDefaultDoctype());
49      }
50  }
