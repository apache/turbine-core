022import org.apache.turbine.TurbineConstants;
023import org.apache.turbine.pipeline.PipelineData;
024import org.apache.turbine.util.RunData;
025import org.apache.velocity.context.Context;
026
027/**
028 * VelocityErrorScreen screen - directs errors at the velocity
029 * error template defined in template.error.
030 *
031 * @author <a href="mailto:gonzalo.diethelm@sonda.com">Gonzalo Diethelm</a>
032 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
033 * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux</a>
034 * @version $Id: VelocityErrorScreen.java 1854797 2019-03-04 20:41:39Z tv $
035 */
036public class VelocityErrorScreen
037    extends VelocityScreen
038{
039    /**
040     * Implement this to add information to the context.
041     *
042     * @param pipelineData Turbine information.
043     * @param context Context for web pages.
044     * @throws Exception a generic exception.
045     */
046    @Override
047    protected void doBuildTemplate(PipelineData pipelineData, Context context)
048            throws Exception
049    {
050        RunData data = pipelineData.getRunData();
051        context.put (TurbineConstants.PROCESSING_EXCEPTION_PLACEHOLDER,
052                data.getStackTraceException().toString());
053        context.put (TurbineConstants.STACK_TRACE_PLACEHOLDER,
054                data.getStackTrace());
055
056        setTemplate(pipelineData, templateError);
057    }
058}
