022import java.io.OutputStreamWriter;
023import java.io.PrintWriter;
024
025import org.apache.turbine.pipeline.PipelineData;
026import org.apache.turbine.util.RunData;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030
031/**
032 * A Screen class for dealing with JSON requests.  Typically you would
033 * extend this class and override the doOutput() method to use it by setting the JSON output into
034 * rundata.setMessage( serialized ).
035 * As convenience you may use inject in your extended class the Turbine service JsonService
036 * Use {@link PlainJSONSecureAnnotatedScreen} if you need the user to be
037 * logged in or having a special role in prior to executing the functions you provide.
038 *
039 * <p>Here is an example from a subclass:
040 *
041 * <code>
042 * 
043 *
044 * public void doOutput(PipelineData pipelineData) throws Exception
045 * {
046 *     RunData data = pipelineData.getRunData();
047 *     JSONStrategy strategy = null;
048 *     
049 *     try
050 *     {
051 *        strategy = new XYStrategy();
052 *        // the result goes into rundata.message
053 *        strategy.execute(data, jsonService);
054 *     }
055 *       catch ( Exception e )
056 *       {
057 *          log.error( "init failed for "+strategy , e);
058 *          String msg = new JSONObject().put("error", e.getMessage()).toString();
059 *          data.setMessage( msg );
060 *       }
061 *     
062 *     super.doOutput(data);
063 * }
064 * </code>
065 *
066 *
067 * @author gk
068 * @version $Id$
069 */
070public class PlainJSONScreen extends RawScreen
071{
072    protected static final String JSON_TYPE = "application/json;charset=utf-8";
073
074    protected final static int BUFFER_SIZE = 4096;
075    
076    static final Logger log = LoggerFactory.getLogger(PlainJSONScreen.class);
077
078    /** Injected service instance */
079    //@TurbineService
080    //protected JsonService jsonService;
081
082    /**
083     * @see org.apache.turbine.modules.screens.RawScreen#getContentType(org.apache.turbine.pipeline.PipelineData)
084     */
085    @Override
086    protected String getContentType(PipelineData pipelineData)
087    {
088        return JSON_TYPE;
089    }
090
091    /**
092     * Output JSON content set into {@link RunData#getMessage()}.
093     *
094     * Encoding is UTF-8. @{@link #JSON_TYPE}: {@value #JSON_TYPE}.
095     *
096     * @param pipelineData The PipelineData object.
097     */
098    @Override
099    protected void doOutput(PipelineData pipelineData) throws Exception
100    {
101        RunData data = pipelineData.getRunData();
102        // read in json!
103        String charset =  "UTF-8"; //request.getCharacterEncoding();
104        
105        String json_res =data.getMessage();
106
107        log.debug( "json_res output: {}", json_res );
108        PrintWriter out = new PrintWriter(
109                new OutputStreamWriter(data.getResponse().getOutputStream(),charset));
110        out.print(json_res.toString());
111        out.flush();
112        out.close();
113    }
114}
