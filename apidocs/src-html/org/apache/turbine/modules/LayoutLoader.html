022import org.apache.turbine.Turbine;
023import org.apache.turbine.pipeline.PipelineData;
024
025/**
026 * The purpose of this class is to allow one to load and execute
027 * Layout modules.
028 *
029 * @author <a href="mailto:mbryson@mont.mindspring.com">Dave Bryson</a>
030 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
031 * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux</a>
032 * @version $Id$
033 */
034public class LayoutLoader
035    extends GenericLoader<Layout>
036{
037    /** The single instance of this class. */
038    private static LayoutLoader instance = new LayoutLoader();
039
040    /**
041     * These ctor's are private to force clients to use getInstance()
042     * to access this class.
043     */
044    private LayoutLoader()
045    {
046        super(Layout.class,
047                () -> Turbine.getConfiguration().getInt(Layout.CACHE_SIZE_KEY,
048                        Layout.CACHE_SIZE_DEFAULT));
049    }
050
051    /**
052     * Attempts to load and execute the external layout.
053     *
054     * @param pipelineData Turbine information.
055     * @param name Name of object that will execute the layout.
056     * @throws Exception a generic exception.
057     */
058    @Override
059    public void exec(PipelineData pipelineData, String name)
060                throws Exception
061    {
062        getAssembler(name).build(pipelineData);
063    }
064
065    /**
066     * The method through which this class is accessed.
067     *
068     * @return The single instance of this class.
069     */
070    public static LayoutLoader getInstance()
071    {
072        return instance;
073    }
074}
