024import java.io.IOException;
025
026import org.apache.logging.log4j.LogManager;
027import org.apache.logging.log4j.Logger;
028import org.apache.turbine.util.RunData;
029import org.apache.turbine.util.TurbineException;
030import org.apache.turbine.util.uri.URIConstants;
031
032/**
033 * This valve is responsible for setting the 'target' property of the RunData.
034 * If it is not already set it attempts to get the target from the request
035 * parameter 'template'. If the parameter is not set, we use the homepage
036 * specified by the configuration property Turbine.TEMPLATE_HOMEPAGE.
037 *
038 * FIXME: The request parameter which determines the template should be
039 *        configurable.
040 *
041 * @author <a href="mailto:james@jamestaylor.org">James Taylor</a>
042 * @author <a href="mailto:peter@courcoux.biz">Peter Courcoux</a>
043 */
044public class DetermineTargetValve implements Valve
045{
046    private static final Logger log
047        = LogManager.getLogger(DetermineTargetValve.class);
048
049    /**
050     * @see org.apache.turbine.pipeline.Valve#invoke(PipelineData, ValveContext)
051     */
052    @Override
053    public void invoke(PipelineData pipelineData, ValveContext context)
054        throws IOException, TurbineException
055    {
056        RunData runData = pipelineData.getRunData();
057        if (!runData.hasScreen())
058        {
059            String target = runData.getParameters().getString(URIConstants.CGI_SCREEN_PARAM);
060
061            if (target != null)
062            {
063                runData.setScreen(target);
064                log.debug("Set screen target from request parameter");
065            }
066            else
067            {
068                                log.debug("No target screen");
069            }
070        }
071
072        log.debug("Screen Target is now: {}", runData::getScreen);
073
074        // Pass control to the next Valve in the Pipeline
075        context.invokeNext(pipelineData);
076    }
077}
