024import java.io.IOException;
025import java.io.UnsupportedEncodingException;
026
027import javax.servlet.http.HttpServletRequest;
028
029import org.apache.logging.log4j.LogManager;
030import org.apache.logging.log4j.Logger;
031import org.apache.turbine.Turbine;
032import org.apache.turbine.util.LocaleUtils;
033import org.apache.turbine.util.TurbineException;
034
035/**
036 * Set default encoding of the request. The default behavior is to respond
037 * with the charset that was requested. If the configuration sets a property named
038 * "locale.override.charset", the output encoding will always be set to its value,
039 * no matter what the input encoding is.
040 *
041 * This valve must be situated in the pipeline before any access to the
042 * {@link org.apache.fulcrum.parser.ParameterParser} to take effect.
043 *
044 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
045 */
046public class DefaultSetEncodingValve
047    implements Valve
048{
049    private static final Logger log = LogManager.getLogger(DefaultSetEncodingValve.class);
050
051    /**
052     * @see org.apache.turbine.pipeline.Valve#invoke(PipelineData, ValveContext)
053     */
054    @Override
055    public void invoke(PipelineData pipelineData, ValveContext context)
056        throws IOException, TurbineException
057    {
058        HttpServletRequest req = pipelineData.get(Turbine.class, HttpServletRequest.class);
059
060        // If the servlet container gives us no clear indication about the
061        // encoding of the contents, set it to our default value.
062        String requestEncoding = req.getCharacterEncoding();
063
064        if (requestEncoding == null)
065        {
066            requestEncoding = LocaleUtils.getDefaultInputEncoding();
067            log.debug("Changing Input Encoding to {}", requestEncoding);
068
069            try
070            {
071                req.setCharacterEncoding(requestEncoding);
072            }
073            catch (UnsupportedEncodingException uee)
074            {
075                throw new TurbineException("Could not change request encoding to " + requestEncoding, uee);
076            }
077        }
078
079        // Copy encoding charset to RunData to set a reasonable default for the response
080        String outputEncoding = LocaleUtils.getOverrideCharSet();
081        if (outputEncoding == null)
082        {
083            outputEncoding = requestEncoding;
084        }
085
086        pipelineData.getRunData().setCharSet(outputEncoding);
087
088        // Pass control to the next Valve in the Pipeline
089        context.invokeNext(pipelineData);
090    }
091}
