024/**
025 * This interface marks a class that has a retrieve(String) method.
026 *
027 * @author <a href="mailto:jmcnally@collab.net">John D. McNally</a>
028 * @version $Id$
029 *
030 * @deprecated cft. deprecation comment in {@link OMTool}.
031 */
032@Deprecated
033public interface Retriever
034{
035    /**
036     * Gets an object of the same type as the object implementing
037     * this method according to the key.
038     *
039     * @param key the key for the object
040     * @return the object that matches the key
041     * @throws Exception if something goes wrong
042     */
043    Object retrieve(String key)
044        throws Exception;
045}
