024import java.util.Hashtable;
025import java.util.Iterator;
026import java.util.List;
027
028/**
029 * Used for adding and accessing messages that relate to a specific form and field. Allows to query for messages by form
030 * name and field name. Used together with FormMessage class.
031 *
032 * @author <a href="mailto:neeme@one.lv">Neeme Praks</a>
033 * @version $Id: FormMessages.java 1854688 2019-03-03 10:36:42Z tv $
034 */
035public class FormMessages
036{
037    private final Hashtable<String, List<String>> forms_messages;
038
039    private final Hashtable<String, List<String>> fields_messages;
040
041    private final Hashtable<String, List<String>> messages_fields;
042
043    private final Hashtable<String, List<String>> forms_fields;
044
045    /**
046     * Constructor.
047     */
048    public FormMessages()
049    {
050        forms_messages = new Hashtable<String, List<String>>();
051        fields_messages = new Hashtable<String, List<String>>();
052        messages_fields = new Hashtable<String, List<String>>();
053        forms_fields = new Hashtable<String, List<String>>();
054    }
055
056    /**
057     * Sets a message for a field of a form. The message is given as a long representing a return code.
058     *
059     * @param formName A String with the form name.
060     * @param fieldName A String with the field name.
061     * @param returnCode A long with the return code.
062     */
063    public void setMessage( String formName, String fieldName, long returnCode )
064    {
065        setMessage( formName, fieldName, String.valueOf( returnCode ) );
066    }
067
068    /**
069     * Sets a message for a field of a form. The message is given as a String.
070     *
071     * @param formName A String with the form name.
072     * @param fieldName A String with the field name.
073     * @param messageName A String with the message.
074     */
075    public void setMessage( String formName, String fieldName, String messageName )
076    {
077        String formFieldName = formName + "-" + fieldName;
078        addValue( forms_messages, formName, messageName );
079        addValue( fields_messages, formFieldName, messageName );
080        addValue( messages_fields, messageName, formFieldName );
081        addValue( forms_fields, formName, formFieldName );
082    }
083
084    /**
085     * Adds a pair key/value to a table, making sure not to add duplicate keys.
086     *
087     * @param table A Hashtable.
088     * @param key A String with the key.
089     * @param value A String with value.
090     */
091    private void addValue( Hashtable<String, List<String>> table, String key, String value )
092    {
093        List<String> values;
094
095        if ( !table.containsKey( key ) )
096        {
097            values = new ArrayList<String>();
098            values.add( value );
099            table.put( key, values );
100        }
101        else
102        {
103            values = table.get( key );
104            if ( !values.contains( value ) )
105            {
106                values.add( value );
107            }
108        }
109    }
110
111    /**
112     * Gets a pair key/value from a table.
113     *
114     * @param table A Hashtable.
115     * @param key A String with the key.
116     * @return A List with the pair key/value, or null.
117     */
118    private final List<String> getValues( Hashtable<String, List<String>> table, String key )
119    {
120        return table.get( key );
121    }
122
123    /**
124     * Gets all form messages for a given form.
125     *
126     * @param formName A String with the form name.
127     * @return A FormMessage[].
128     */
129    public FormMessage[] getFormMessages( String formName )
130    {
131        List<String> messages, fields;
132        String messageName, fieldName;
133        messages = getValues( forms_messages, formName );
134        if ( messages != null )
135        {
136            FormMessage[] result = new FormMessage[messages.size()];
137            for ( int i = 0; i < messages.size(); i++ )
138            {
139                result[i] = new FormMessage( formName );
140                messageName = messages.get( i );
141                result[i].setMessage( messageName );
142                fields = getValues( messages_fields, messageName );
143                for ( int j = 0; j < fields.size(); j++ )
144                {
145                    fieldName = fields.get( j );
146                    if ( formHasField( formName, fieldName ) )
147                    {
148                        result[i].setFieldName( fieldName );
149                    }
150                }
151            }
152            return result;
153        }
154        return null;
155    }
156
157    /**
158     * Get form messages for a given form and field.
159     *
160     * @param formName A String with the form name.
161     * @param fieldName A String with the field name.
162     * @return A FormMessage[].
163     */
164    public FormMessage[] getFormMessages( String formName, String fieldName )
165    {
166        String key = formName + "-" + fieldName;
167
168        List<String> messages = getValues( fields_messages, key );
169        String messageName;
170
171        if ( messages != null )
172        {
173            FormMessage[] result = new FormMessage[messages.size()];
174            for ( int i = 0; i < messages.size(); i++ )
175            {
176                result[i] = new FormMessage( formName, fieldName );
177                messageName = messages.get( i );
178                result[i].setMessage( messageName );
179            }
180            return result;
181        }
182        return null;
183    }
184
185    /**
186     * Check whether a form as a field.
187     *
188     * @param formName A String with the form name.
189     * @param fieldName A String with the field name.
190     * @return True if form has the field.
191     */
192    private boolean formHasField( String formName, String fieldName )
193    {
194        List<String> fields = getValues( forms_fields, formName );
195        for ( Iterator<String> iter = fields.iterator(); iter.hasNext(); )
196        {
197            if ( fieldName.equals( iter.next().toString() ) )
198            {
199                return true;
200            }
201        }
202        return false;
203    }
204}
